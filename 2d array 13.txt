#include <stdio.h>

int main() {
    int r, c;
    scanf("%d %d", &r, &c);

    int a[100][100];

    
    for(int i = 0; i < r; i++) {
        for(int j = 0; j < c; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    int maxSum = -1000000;
    int maxIndex = 0;

   
    for(int j = 0; j < c; j++) {
        int sum = 0;
        for(int i = 0; i < r; i++) {
            sum += a[i][j];
        }

        if(sum > maxSum) {
            maxSum = sum;
            maxIndex = j;
        }
    }

    printf("%d", maxIndex);

    return 0;
}
