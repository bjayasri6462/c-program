#include <stdio.h>

int main() {
    int n;

    scanf("%d", &n);
    int arr[n];

    
    for(int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

   
    for(int i = 0; i < n; i++) {
        int unique = 1;

        for(int j = 0; j < i; j++) {
            if(arr[i] == arr[j]) {
                unique = 0;
                break;
            }
        }

        if(unique) {
            printf("%d ", arr[i]);
        }
    }

    return 0;
}
