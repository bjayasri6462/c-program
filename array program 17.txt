#include <stdio.h>

int main() {
    int n, i;
    scanf("%d", &n);

    int a[n];

    for(i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    for(i = 0; i < n/2; i++) {
        if(a[i] != a[n - 1 - i]) {
            printf("No");
            return 0;
        }
    }

    printf("Yes");
    return 0;
}
