#include <stdio.h>

int main() {
    char str [100];
    int i,flag =0;
    scanf("%s",str);
    for (i=0;str[i] !='\0';i++)
    {
        if(str[i]>='A'&& str [i]<='Z')
        {
            flag =1;
            break;
        }
    }
    if(flag==1)
    printf("valid");
    else
    printf("invalid");
    return 0;
}
