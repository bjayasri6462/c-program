#include <stdio.h>

int main() {
    char str[200];
    int i = 0, spaces = 0;

    fgets(str, sizeof(str), stdin);

    while(str[i] != '\0') {
        if(str[i] == ' ')
            spaces++;
        i++;
    }

    if(i > 1)
        printf("%d", spaces + 1);
    else
        printf("0");

    return 0;
}
