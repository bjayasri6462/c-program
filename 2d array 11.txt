#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);

    int a[R][C];
    for(int i = 0; i < R; i++) {
        for(int j = 0; j < C; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    int sortedCount = 0;

    for(int i = 0; i < R; i++) {
        int isSorted = 1;
        for(int j = 0; j < C - 1; j++) {
            if(a[i][j] > a[i][j+1]) {
                isSorted = 0;
                break;
            }
        }
        if(isSorted)
            sortedCount++;
    }

    printf("%d", sortedCount);

    return 0;
}
