#include <stdio.h>

int main() {
    int n, i, j, count = 0;
    scanf("%d", &n);

    int a[n];

    for(i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    for(i = 0; i < n; i++) {
        int found = 0;

        for(j = 0; j < i; j++) {
            if(a[i] == a[j]) {
                found = 1;
                break;
            }
        }

        if(found == 0) {
            count++;
        }
    }

    printf("%d", count);

    return 0;
}
