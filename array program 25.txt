#include <stdio.h>
#include <limits.h>

int main() {
    int n;
    scanf("%d", &n);

    int a[n];

    for(int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    int max1 = INT_MIN, max2 = INT_MIN;
    int min1 = INT_MAX, min2 = INT_MAX;

    for(int i = 0; i < n; i++) {

        if(a[i] > max1) {
            max2 = max1;
            max1 = a[i];
        } 
        else if(a[i] > max2) {
            max2 = a[i];
        }

        if(a[i] < min1) {
            min2 = min1;
            min1 = a[i];
        } 
        else if(a[i] < min2) {
            min2 = a[i];
        }
    }

    int p1 = max1 * max2;
    int p2 = min1 * min2;

    if(p1 > p2)
        printf("%d", p1);
    else
        printf("%d", p2);

    return 0;
}
